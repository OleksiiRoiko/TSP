# CONFIG_REFERENCE (TSP-GNN)
# Updated to match current config.py implementation

# You can set `base: path/to/base.yaml` to load a base config and override only the fields you need.
# You can use `{exp_id}` in string fields; it will be replaced with train.exp_id.
# See configs/_template.yaml for a minimal per-experiment override template.
# Note: sections can be omitted from a YAML file; defaults from config.py will apply.

[generate]
out_root          : str   | base folder for synthetic dataset (train/val/test)
n_list            : [int] | list of N (number of nodes) to generate (>=3)
per_size_train    : int   | number of graphs per N for train
per_size_val      : int   | number of graphs per N for val
per_size_test     : int   | number of graphs per N for test
tour_solver       : str   | auto | concorde | elkai | nn2opt
elkai_frac        : float | fraction in [0,1] of graphs solved by elkai (else NN+2opt)
concorde_cmd      : str   | concorde binary name/path (only for tour_solver=concorde)
concorde_scale    : int   | coordinate scale for TSPLIB EUC_2D (e.g., 10000)
concorde_timeout_sec : int | per-instance timeout for concorde
concorde_keep_tmp : bool  | keep temp concorde files under runs/tmp/concorde
dist_names        : [str] | list of distribution names (e.g., ["uniform","clustered","ring","grid_jitter"])
dist_probs        : [float] | sampling probs aligned to dist_names (normalized internally)
seed              : int   | RNG seed for reproducibility

[tsplib]
raw_root          : str   | folder containing downloaded .tsp/.opt.tour files
out_root          : str   | processed .npz output folder
download          : bool  | auto-download missing instances
names             : [str] | list of TSPLIB instance names; e.g., ["berlin52", "eil76"]

[train]
exp_id            : str   | experiment label (used in filenames/logs)
train_root        : str   | root of training .npz (e.g., runs/data/synthetic/train)
val_root          : str   | root of validation .npz
epochs            : int   | training epochs
batch_size        : int   | graphs per batch (edges are concatenated per batch)
lr                : float | learning rate
weight_decay      : float | L2 regularization for Adam
lr_scheduler      : str   | "none" or "plateau" (ReduceLROnPlateau)
lr_factor         : float | LR decay factor (plateau scheduler)
lr_patience       : int   | epochs without improvement before LR decay
lr_min            : float | minimum LR for scheduler
early_stop        : bool  | enable early stopping on val loss
early_patience    : int   | epochs without improvement before stopping
early_min_delta   : float | minimum improvement to reset patience
seed              : int   | RNG seed
model_name        : str   | "edge_mlp" (shallow) or "edge_mlp_deep" (deeper)
hidden            : int?  | hidden layer width (override model default)
dropout           : float?| dropout rate (override model default)
depth             : int   | number of hidden layers in the edge MLP
# Note: training uses complete graphs (no sparse candidates).

[eval]
model_path        : str   | checkpoint .pt path or a latest.json pointer
data_roots        : [str] | evaluate multiple datasets in one run
device            : str   | "cpu" or "cuda"
save_json         : str?  | "auto" to store under run dir, or explicit path (null = do not save)
run_twoopt        : bool  | run 2-opt after greedy decode
seed              : int   | rng seed for reproducibility

[visualize]
targets           : list  | list of {npz_dir, limit?, out_dir?, mode?} for multi-dataset runs
model             : str   | checkpoint .pt or latest.json pointer (predict mode)
device            : str   | "cpu" or "cuda"
out_dir           : str   | "auto" to store under run dir, or explicit path
figsize           : [w,h] | inches
dpi               : int   | dots per inch

[qa]
root              : str   | root folder to scan recursively for .npz to check
check_gt          : bool  | validates GT is a Hamiltonian cycle (permutation 0..n-1)
lengths           : bool  | verifies stored normalized length matches recomputed
require_label_source : str? | if set, enforce label_source for synthetic files (e.g., "concorde")
check_concorde    : bool  | validate concorde-specific fields/lengths when label_source=concorde
csv               : str?  | optional CSV report path (null = don't save)
