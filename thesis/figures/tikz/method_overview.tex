% method_overview.tex
% Celkový návrh experimentálneho postupu
% Ľavá vetva: trénovanie na syntetických dátach (heuristické + Concorde riešenia)
% Pravá vetva: hodnotenie na TSPLIB inštanciách
% Spodok: spoločná záverečná analýza a porovnanie
% Použitie: \resizebox{0.96\textwidth}{!}{\input{figures/tikz/method_overview}}
\begin{tikzpicture}[
    font=\small,
    train/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=3.4cm, minimum height=0.78cm,
        inner sep=4pt, fill=blue!8, draw=blue!45
    },
    refbox/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=2.8cm, minimum height=0.78cm,
        inner sep=4pt, fill=orange!10, draw=orange!50
    },
    evalbox/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=3.4cm, minimum height=0.78cm,
        inner sep=4pt, fill=green!8, draw=green!50
    },
    finalbox/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=9.0cm, minimum height=0.90cm,
        inner sep=4pt, fill=gray!12, draw=gray!55
    },
    %% Štandardná šípka — rovnaký hrúbka a štýl vo celom diagrame
    arrow/.style={-{Latex[length=2.5mm, width=1.8mm]}, thick},
    %% Prerušovaná šípka pre zdieľané modely
    darrow/.style={-{Latex[length=2.5mm, width=1.8mm]}, thick, dashed, draw=gray!55}
]

%% ══════════════════════════════════════════════════════════════
%% ĽAVÁ VETVA — tréningová  (stĺpec x = 0)
%% ══════════════════════════════════════════════════════════════

\node[train] (synth) at (0, 0)
    {Syntetické inštancie\\ \(n \in \{20,30,\ldots,100\}\)};

\node[train] (ref) at (0, -1.60)
    {Referenčné riešenia};

%% Dve úrovne kvality — symetricky okolo x=0
\node[refbox] (heur) at (-2.0, -3.20)
    {Heuristické\\ (NN + 2-opt)};

\node[refbox] (conc) at (2.0, -3.20)
    {Presné\\ (Concorde)};

\node[train] (trainmod) at (0, -4.80)
    {Trénovanie modelov\\ (MLP, ResMLP, Transformer)};

%% ══════════════════════════════════════════════════════════════
%% PRAVÁ VETVA — hodnotiaca  (stĺpec x = 8.0)
%% Uzly na rovnakých y-úrovniach ako ľavá vetva
%% ══════════════════════════════════════════════════════════════

\node[evalbox] (tsplib) at (8.0, 0)
    {TSPLIB inštancie\\ (14 štandardných)};

\node[evalbox] (norm) at (8.0, -1.60)
    {Normalizácia\\ súradníc};

\node[evalbox] (feats) at (8.0, -3.20)
    {Grafová reprezentácia\\ + príznaky hrán};

\node[evalbox] (modapp) at (8.0, -4.80)
    {Natrénované modely\\ + heuristické bázy};

%% ══════════════════════════════════════════════════════════════
%% SPOLOČNÁ DOLNÁ ČASŤ  (centrované na x = 4.0)
%% ══════════════════════════════════════════════════════════════

\node[train] (decode) at (4.0, -7.00)
    {Dekódovanie + lokálne\\ zlepšenie (2-opt)};

\node[finalbox] (anal) at (4.0, -8.50)
    {Záverečná analýza a porovnanie výsledkov};

%% ══════════════════════════════════════════════════════════════
%% ŠÍPKY — ľavá vetva
%% ══════════════════════════════════════════════════════════════

\draw[arrow] (synth) -- (ref);
\draw[arrow] (ref)   -- (heur.north);
\draw[arrow] (ref)   -- (conc.north);

%% heur/conc → trainmod: T-spoj (symetrický)
%% Oba uzly idú nadol na y=-4.16, tam sa spoja horizontálnou čiarou,
%% z ktorej stredový bod smeruje šípkou na trainmod.north
\coordinate (lhj) at (-2.0, -4.16);
\coordinate (rhj) at ( 2.0, -4.16);
\draw[thick] (heur.south) -- (lhj);
\draw[thick] (conc.south) -- (rhj);
\draw[thick] (lhj)        -- (rhj);
\draw[arrow] (0, -4.16)   -- (trainmod.north);

%% ══════════════════════════════════════════════════════════════
%% ŠÍPKY — pravá vetva
%% ══════════════════════════════════════════════════════════════

\draw[arrow] (tsplib) -- (norm);
\draw[arrow] (norm)   -- (feats);
\draw[arrow] (feats)  -- (modapp);

%% ══════════════════════════════════════════════════════════════
%% ŠÍPKA — zdieľané modely (ľavá → pravá, prerušovaná)
%% ══════════════════════════════════════════════════════════════

\draw[darrow] (trainmod.east) --
    node[above, font=\scriptsize, yshift=1pt] {zdieľané}
    (modapp.west);

%% ══════════════════════════════════════════════════════════════
%% KONVERGENCIA — T-spoj do decode
%% trainmod a modapp idú nadol na y=-5.80, spoja sa horizontálne,
%% stredový bod → šípka na decode.north
%% ══════════════════════════════════════════════════════════════

\coordinate (lj) at (0,   -5.80);
\coordinate (rj) at (8.0, -5.80);
\draw[thick] (trainmod.south) -- (lj);
\draw[thick] (modapp.south)   -- (rj);
\draw[thick] (lj)             -- (rj);
\draw[arrow] (4.0, -5.80)     -- (decode.north);

%% ══════════════════════════════════════════════════════════════
%% ŠÍPKA — decode → záverečná analýza
%% ══════════════════════════════════════════════════════════════

\draw[arrow] (decode) -- (anal);

%% ══════════════════════════════════════════════════════════════
%% HLAVIČKY VETIEV
%% ══════════════════════════════════════════════════════════════

\node[font=\footnotesize\bfseries, text=blue!65]          at (0,   0.70) {Tréningová vetva};
\node[font=\footnotesize\bfseries, text=green!60!black]   at (8.0, 0.70) {Hodnotiaca vetva};

\end{tikzpicture}
