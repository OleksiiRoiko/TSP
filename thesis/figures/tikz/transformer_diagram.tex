% transformer_diagram.tex — horizontálne rozloženie: Fáza 1 vľavo, Fáza 2 vpravo
% Vyžaduje: \usetikzlibrary{arrows.meta,positioning,calc}
% Použitie: \resizebox{0.95\textwidth}{!}{\input{figures/tikz/transformer_diagram}}
\begin{tikzpicture}[
    font=\small,
    node distance=0.55cm,
    ibox/.style  ={draw=gray!50,  fill=gray!8,   rounded corners=3pt,
                   align=center, minimum width=4.0cm, inner sep=5pt},
    abox/.style  ={draw=orange!50,fill=orange!8, rounded corners=3pt,
                   align=center, minimum width=4.0cm, inner sep=5pt},
    cbox/.style  ={draw=green!55!black,fill=green!7,rounded corners=3pt,
                   align=center, minimum width=4.0cm, inner sep=5pt},
    sbox/.style  ={draw=green!55!black,fill=green!7,rounded corners=3pt,
                   align=center, minimum width=1.5cm, inner sep=4pt},
    phibox/.style={draw=gray!50, fill=gray!8,   rounded corners=3pt,
                   align=center, minimum width=1.5cm, inner sep=4pt},
    mbox/.style  ={draw=purple!40,fill=purple!6, rounded corners=3pt,
                   align=center, minimum width=4.0cm, inner sep=5pt},
    obox/.style  ={draw=purple!55,fill=purple!10,rounded corners=3pt,
                   align=center, minimum width=4.0cm, inner sep=5pt},
    arrow/.style ={-{Latex[length=2.5mm,width=1.8mm]},thick,shorten >=3pt},
    garrow/.style={-{Latex[length=2.3mm,width=1.6mm]},thick,
                   draw=green!60!black,shorten >=3pt},
    pline/.style ={thick}
]

%% ══════════════════════════════════════════
%% FÁZA 1 — ľavý stĺpec (vertikálny tok)
%% ══════════════════════════════════════════
\node[font=\small\bfseries, text=orange!70!black]
    (h1) at (0,0) {Fáza~1 — kódovanie vrcholov};

\node[ibox, below=0.30cm of h1] (coords)
    {Súradnice miest \((u_i,\,v_i)\)\\[2pt]
     \scriptsize pre všetky vrcholy inštancie};

\node[abox, below=of coords] (embed)
    {Lineárna projekcia\\[2pt]
     \scriptsize súradnice \(\to\) vnútorná reprezentácia};

\node[abox, below=of embed] (attn)
    {Mechanizmus vlastnej pozornosti\\[2pt]
     \scriptsize opakuje sa \(L\)-krát};

\node[cbox, below=of attn] (ctx)
    {Kontextové reprezentácie \(z_i\)\\[2pt]
     \scriptsize poloha + vzťah ku všetkým vrcholom};

\draw[arrow] (coords.south) -- (embed.north);
\draw[arrow] (embed.south)  -- (attn.north);
\draw[arrow] (attn.south)   -- (ctx.north);

%% ══════════════════════════════════════════
%% FÁZA 2 — pravý stĺpec
%% vstupné uzly sú zarovnané s uzlmi Fázy 1
%% ══════════════════════════════════════════

% Nadpis Fázy 2 — horizontálne zarovnaný s h1
\node[font=\small\bfseries, text=purple!65!black]
    (h2) at ($(h1)+(9.2,0)$) {Fáza~2 — skórovanie hrán};

% Vstupné uzly: zi ↔ embed, phi ↔ attn, zj ↔ ctx (rovnaké y-súradnice)
\node[sbox]    (zi)  at ($(embed)+(5.4,0)$)
    {\(z_i\)\\[2pt]\scriptsize kontext \(i\)};

\node[phibox]  (phi) at ($(attn)+(5.4,0)$)
    {\(\phi_{ij}\)\\[2pt]\scriptsize príznaky};

\node[sbox]    (zj)  at ($(ctx)+(5.4,0)$)
    {\(z_j\)\\[2pt]\scriptsize kontext \(j\)};

% Merge box — vpravo od vstupných uzlov (zarovnaný na phi/stred)
\node[mbox] (merge) at ($(phi)+(5.2,0)$)
    {Zlúčenie + výstupná sieť\\[2pt]
     \scriptsize dve lineárne vrstvy s aktiváciou GELU};

% Score box — pod merge
\node[obox, below=0.50cm of merge] (score)
    {Skóre hrany \(s_{ij} \in \mathbb{R}\)\\[2pt]
     \scriptsize miera príslušnosti hrany k trase};

%% ══════════════════════════════════════════
%% PRECHOD  Fáza 1 → Fáza 2
%% ctx.east ──► vetva ──► zi / phi / zj
%%
%%   ctx ──── bx ─┬──► zi
%%               │
%%               ├──► phi
%%               │
%%               └──► zj
%% ══════════════════════════════════════════

% Bod vetvenia: 45 % cesty medzi ctx.east a zj.west (rovnaká y-úroveň)
\coordinate (bx) at ($(ctx.east)!0.45!(zj.west)$);

% Horizontálna čiara: ctx.east → bod vetvenia
\draw[pline, draw=green!60!black] (ctx.east) -- (bx);

% Vertikálna hrebeňová čiara: od úrovne zi po úroveň zj
\draw[pline, draw=green!60!black] (bx |- zi.west) -- (bx |- zj.west);

% Horizontálne šípky: vetva → zi, phi, zj
\draw[garrow] (bx |- zi.west)  -- (zi.west);
\draw[garrow] (bx |- phi.west) -- (phi.west);
\draw[garrow] (bx |- zj.west)  -- (zj.west);

%% ══════════════════════════════════════════
%% ŠÍPKY FÁZY 2:  zi/phi/zj → T-spoj → merge → score
%% ══════════════════════════════════════════

% T-spoj x: stred medzi zi.east a merge.west (na rovnakej y-úrovni)
\coordinate (tj) at ($(zi.east)!0.50!(merge.west |- zi.east)$);

% Horizontálne čiary: každý vstup → T-spoj
\draw[pline] (zi.east)  -- (tj |- zi.east);
\draw[pline] (phi.east) -- (tj |- phi.east);
\draw[pline] (zj.east)  -- (tj |- zj.east);

% Vertikálna T-lišta spájajúca zi, phi, zj pri x=tj
\draw[pline] (tj |- zi.east) -- (tj |- zj.east);

% Šípka z T-spoja (na výške phi = stred) do merge.west
\draw[arrow] (tj |- phi.east) -- (merge.west);

% Šípka merge → score
\draw[arrow] (merge.south) -- (score.north);

%% ══════════════════════════════════════════
%% Vertikálna oddeľovacia čiara medzi fázami
%% ══════════════════════════════════════════
\draw[gray!35, line width=0.5pt, dashed]
    (3.2, 0.40) -- (3.2, -7.20);

\end{tikzpicture}
